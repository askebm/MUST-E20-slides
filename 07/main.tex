\begin{frame}[t]
	\frametitle{Search Methods}
	\framesubtitle{Gaussian Mixture}


	\begin{itemize}
		\item Model\\
			\[
				f _{X} \left( X \right)  =
				\sum^{K}_{k=1} 
				p_k f _{Y_k}  \left( x \right) ,
				\hspace{1cm}
				x = \left[ x_1, \cdots, x_p \right] ^{T}
			.\] 
			\pause
		\item Estimate 
			\[
				L  \left( \{ p_k \} ^{K}_{k=1},
					\{ \mu_k \}^{K}_{k=1},
					\{ \Sigma_{k} \}^{K}_{k=1}
				\right) 
			.\] 
			\pause
		\item Evaluation
			\[
				\begin{aligned}
					AIC &= 2 \log L _{\max} - 2
					\left( K \cdot  \left[ 1 + \frac{p \left( p+3 \right) }{2}  \right] -1  \right) \\
					BIC &= 2 \log L _{\max} -
					\left(  K \cdot  \left[ 1 + \frac{p \left( p+3 \right) }{2}  \right] -1  \right) 
					\log n
				\end{aligned}
			.\] 
    \item Assign Clusters
	\end{itemize}


\end{frame}
