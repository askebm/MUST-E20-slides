\begin{frame}[t]
    \frametitle{Sample Based}

    \begin{align*}
    X = \begin{bmatrix}
			X_{11} & \hdots & X_{1p} \\
        \vdots &  \ddots & \vdots  \\
				X_{j1} & \hdots & X_{jp} \\
    \end{bmatrix}
    \end{align*}

    \begin{align*}
        \hat{\Sigma} = S = \frac{1}{n-1} \sum^{n}_{j=1} (X_j - \bar{X} )(X_j - \bar{X} )^T
    \end{align*}
   \begin{align*}
    \frac{ \sum^{r}_{i=1} \hat{\lambda_i}  }{ \sum^{p}_{j=1} \hat{\lambda_j}  }
   \end{align*}
    

    
\end{frame}
